#### Abfrage von Wordlinks: Wort;LinkID;Link
SELECT word.word,links.id,links.link
FROM wordlinks, word,links
WHERE wordlinks.id_word=word.id 
AND wordlinks.id_link=links.id


#### Abfrage der am häufigsten verwendeten Wörtern
SELECT word.word AS Wort, COUNT(*) AS Anzahl 
FROM wordlinks, word 
WHERE wordlinks.id_word=word.id 
GROUP BY wordlinks.id_word 
ORDER BY Anzahl DESC


#### Sucht die längsten Worte in der DB zusammen mit der website, auf der es vorkommt
SELECT LENGTH(word.word) AS Length, word.word ,links.link
FROM word, wordlinks, links
WHERE word.id=wordlinks.id_word AND
	links.id=wordlinks.id_link
ORDER BY Length DESC
